class Solution:
    def divide(self, dividend, divisor):
        """
        :type dividend: int
        :type divisor: int
        :rtype: int
        """
        positive = (dividend>0) is (divisor>0)
        dividend, divisor = abs(dividend),abs(divisor)
        output = 0
        counter = 1
        tmp = divisor
        while dividend>=divisor:
            dividend-=tmp
            output += counter
            counter<<=1
            tmp<<=1
            if dividend<=tmp:
                tmp=divisor
                counter = 1
        if not positive:
            output = -output
        return max(min(output,(2**31)-1),-2**31)
