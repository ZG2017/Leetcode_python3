class Solution:
    def generate(self, numRows):
        """
        :type numRows: int
        :rtype: List[List[int]]
        """
        res = []
        for i in range(1,numRows+1):
            res.append([1]*i)
        for i in range(1,numRows):
            index = 1
            while index < len(res[i])-1:
                res[i][index] = res[i-1][index-1]+res[i-1][index]
                index += 1
        return res
