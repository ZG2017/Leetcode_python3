class Solution:
    def findLength(self, A: List[int], B: List[int]) -> int:
        dp = [[0]*(len(A)+1) for i in range(len(B)+1)]
        res = 0
        for i in range(1, len(A)+1):
            for j in range(1, len(B)+1):
                if A[i-1] == B[j-1]:
                    dp[i][j] = dp[i-1][j-1] + 1
                    res = max(dp[i][j], res)
                else:
                    dp[i][j] = 0
        return res
