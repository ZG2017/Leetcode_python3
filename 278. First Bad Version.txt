class Solution:
    def firstBadVersion(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n == 0:
            return None
        if not isBadVersion(n):
            return None
        p2 = n
        p1 = 1
        while p2 >= p1:
            tmp = int((p1+p2)/2)
            if isBadVersion(tmp):
                p2 = tmp-1
            else:
                p1 = tmp+1
        return max(p1,p2)
