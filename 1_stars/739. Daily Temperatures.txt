stack

class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
        res = [0]*len(temperatures)
        stack = []
        for idx in range(len(temperatures)):
            if not stack:
                stack.append(idx)
                continue
            while stack and temperatures[idx] > temperatures[stack[-1]]:
                cur_idx = stack.pop()
                res[cur_idx] = idx - cur_idx
            stack.append(idx)
        return res
