# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def mid(self, root):
        if not root:
            return 
        self.mid(root.left)
        self.holder.append(root.val)
        self.mid(root.right)

    def kthLargest(self, root: TreeNode, k: int) -> int:
        self.holder = []
        self.mid(root)
        return self.holder[-k]
