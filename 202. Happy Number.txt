class Solution:
    def isHappy(self, n):
        """
        :type n: int
        :rtype: bool
        """
        holder = {n:1}
        while n != 1:
            n = sum(map(lambda x:int(x)**2,list(str(n))))
            if n not in holder:
                holder[n] = 1
            else:
                return False
        return True
